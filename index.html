Dependency injection

define(function() {
  return {
    name: "duck",
    speak: function() {
      console.log("quack quack");
    }
  };
});

Dependency injection

require(["duck"], function(duck) {
  console.log("What does a " + duck.name + " do?");
  duck.speak();
});

Dependency injection

define(["duck", "pig"], function(duck, pig) {
  return {
    names: [duck.name, pig.name],
    commotion: function() {
      duck.speak();
      pig.speak();
    }
  };
});
